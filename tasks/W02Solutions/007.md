```c++
int removeDuplicates(vector& nums)
{
int i=1,j=1,cnt=1;
int n=nums.size();
while(j<n)
{
if(nums[j]!=nums[j-1])
{
nums[i]=nums[j];
i++;
cnt++;
}
j++;
}
return cnt;
}
```
