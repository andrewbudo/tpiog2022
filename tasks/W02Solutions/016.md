  if(s.length() == 0) return 0;

    for(char c:s){
        
        if(isalpha(c) && !isDigit){
            return 0;
        }else if(isalpha(c) && isDigit){
            break;
        }
        
        if(isspace(c) && isDigit){
            break;
        }
        if(isspace(c) && isZero){
            break;
        }
        
        if(isspace(c) && !isDigit && isSign){
            return 0;
        }
        
        
        if(isdigit(c)){
            if(!isSign) isSign = true;

            if(c == '0'){
                if(isDigit){
                    isZero = true;
                    v += c;
                }else if(!isDigit){
                    isZero = true;
                }
            }else{
                isDigit = true;
                v += c;
            }
        } 
        
        
        if (c == '+' && !isSign){
            v += c;
            isSign = true;
        }else if(c == '+' && isSign){
            //return 0;
            break;
        }
        
        
        if (c == '-' && !isSign){
            v += c;
            isSign = true;
        }else if(c == '-' && isSign){
            //return 0;
            break;
        }
        
        
        if (c == '.' && isDigit){
            break;
        }else if(c == '.' && !isDigit){
            return 0;
        }
        
        
    }

    if(isDigit){
        s = v;
        if(s.length() > 18) s.erase(18, s.length()-18);
        long long x = stoll(s);
        if(x>2147483647) x = 2147483647;
        if(x<-2147483648) x = 2147483648;
        int y = (int)x;
        return y;
    }else{
        return 0;
    }
}
