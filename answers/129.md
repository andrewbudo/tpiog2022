**129.Может ли конструктор быть виртуальным? А деструктор?**  
Конструктор не может быть виртуальным, а деструктор может.  
Пример виртуального деструктора:
```c++
#include <cstdlib>
#include <iostream>
using std::cout;
using std::endl;
class A {
public:
    A() { cout << "A()" << endl; }
    virtual ~A() { cout << "~A()" << endl; }
};
class B : public A {
public:
    B() { cout << "B()" << endl; }
    ~B() { cout << "~B()" << endl; }
};
int main()
{
    A * pA = new B;
    delete pA;
    return EXIT_SUCCESS;
}
```
