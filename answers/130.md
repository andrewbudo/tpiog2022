 **130.Как виртуальные функции влияют на размер класса?**  
К размеру класса добавляется размер виртуальной таблицы функций
```c++
#include <iostream>
#include <iomanip>
using namespace std;
class A
{
    void funcA();
    int x;
};
class B
{
    virtual void funcB();
    int y;
};
#pragma pack(1)
class C
{
    virtual void funcB();
    int y;
};
int main(int argc, const char * argv[])
{
    cout << sizeof(A) << endl;
    cout << sizeof(B) << endl;
    cout << sizeof(C) << endl;
}
```
